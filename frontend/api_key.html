Available Endpoints — AlumniHub Collection
GET /backend/alumni.php Ambil semua data alumni
POST /backend/alumni.php Tambah data alumni baru
GET /backend/dashboard_v2_api.php Data dashboard agregat
POST /backend/auth.php?action=login Login pengguna
POST /backend/auth.php?action=register Registrasi pengguna baru 
# GET — Ambil semua data alumni
curl -X GET http://localhost/integrasi_data/backend/alumni.php \
  -H "Authorization: Bearer <API_KEY>" \
  -H "Content-Type: application/json"

# POST — Tambah data alumni baru
curl -X POST http://localhost/integrasi_data/backend/alumni.php \
  -H "Authorization: Bearer <API_KEY>" \
  -H "Content-Type: application/json" \
  -d '{
    "kode_responden_asli": "R-001",
    "tahun_lulus": "2024",

    "jenjang": "D3",

    "lama_tunggu_bulan": "2"
  }'
